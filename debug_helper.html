Admin.html 布局诊断

请按 F12 打开开发者工具，点击 Elements 标签，然后：

  1. 按 Ctrl+F 搜索 categoriesContainer
  2. 点击该元素，看右侧 Styles 面板显示的 CSS
  3. 截图或告诉我 computed 的 width、margin-left、padding-left 值

或者在 Console 里执行以下代码，把输出结果发给我：

const cc = document.getElementById('categoriesContainer');
if (cc) {
  const rect = cc.getBoundingClientRect();
  const style = window.getComputedStyle(cc);
  console.log('categoriesContainer:');
  console.log('  rect:', JSON.stringify({left:rect.left, top:rect.top, width:rect.width, height:rect.height}));
  console.log('  marginLeft:', style.marginLeft);
  console.log('  paddingLeft:', style.paddingLeft);
  console.log('  width:', style.width);
  
  const parent = cc.parentElement;
  const parentRect = parent.getBoundingClientRect();
  const parentStyle = window.getComputedStyle(parent);
  console.log('parent (#keywordsTab):');
  console.log('  rect:', JSON.stringify({left:parentRect.left, width:parentRect.width}));
  console.log('  paddingLeft:', parentStyle.paddingLeft);
} else {
  console.log('categoriesContainer NOT FOUND');
}
